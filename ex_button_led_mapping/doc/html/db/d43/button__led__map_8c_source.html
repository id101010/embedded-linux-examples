112  * sysfs_gpio_handler - handle GPIO operations
113  ***************************************************************************

115 int sysfs_gpio_handler(uint8_t function, uint32_t gpio, char *val)
116 {
117  char path_str[MAX_PATH_STR];
118  char strBuf[MAX_STR_BUF];
119  uint8_t oflags = 0;
120  int32_t fd;
121  uint32_t len;
122  uint8_t inval;
123 
124  /* Determine open flags based on function */
125  switch (function) {
126  case GPIO_SYSFS_EXPORT:
127  snprintf(path_str, sizeof(path_str), SYSFS_PATH"export");
128  oflags=O_WRONLY;
129  break;
130 
131  case GPIO_SYSFS_UNEXPORT:
132  snprintf(path_str, sizeof(path_str), SYSFS_PATH"unexport");
133  oflags=O_WRONLY;
134  break;
135 
136  case GPIO_SYSFS_SET_DIRECTION:
137  snprintf(path_str, sizeof(path_str), SYSFS_PATH"gpio%d/direction", gpio);
138  oflags=O_WRONLY;
139  break;
140 
141  case GPIO_SYSFS_SET_VALUE:
142  snprintf(path_str, sizeof(path_str), SYSFS_PATH"gpio%d/value", gpio);
143  oflags=O_WRONLY;
144  break;
145 
146  case GPIO_SYSFS_GET_DIRECTION:
147  case GPIO_SYSFS_GET_VALUE:
148  snprintf(path_str, sizeof(path_str), SYSFS_PATH"gpio%d/value", gpio);
149  oflags=O_RDONLY;
150  break;
151 
152  default:
153  printf("File operation flag not defined\n");
154  }
155 
156  /* Open the pseudo file given its path and open flags */
157 #ifdef DEBUG
158  printf("open:%s\n", path_str);
159 #endif
160 
161  fd = open(path_str, oflags);
162  if (fd < 0) {
163  perror(path_str);
164  return fd;
165  }
166 
167  /* File operations r/w on the opened file */
168  switch (function) {
169  case GPIO_SYSFS_EXPORT:
170  case GPIO_SYSFS_UNEXPORT:
171  len = snprintf(strBuf, sizeof(strBuf), "%d", gpio);
172 #ifdef DEBUG
173  printf("exp/unexp:%s\n", strBuf);
174 #endif
175  write(fd, strBuf, len);
176  break;
177 
178  case GPIO_SYSFS_SET_DIRECTION:
179 #ifdef DEBUG
180  printf("write dir:%s\n", val);
181 #endif
182  write(fd, val, strlen(val)+1);
183  break;
184 
185  case GPIO_SYSFS_SET_VALUE:
186 #ifdef DEBUG
187  printf("write val:%s\n", val);
188 #endif
189  write(fd, val, strlen(val)+1);
190  break;
191 
192  case GPIO_SYSFS_GET_DIRECTION:
193  break;
194 
195  case GPIO_SYSFS_GET_VALUE:
196  read(fd, &inval, 1);
197 #ifdef DEBUG
198  printf("read val:%c\n", inval);
199 #endif
200  *val = inval;
201  break;
202 
203  default:
204  printf("function not defined\n");
205  }
206  close(fd);
207  return 0;
208 }
